;Ex30: Design a function that, given a list of booleans,
;returns a list with each boolean reversed 
;(e.g. (neg-list (list true false true)) => (list false true false))


;;DATA DEFINITIONS
;;neg-list List -> List
;;Given: List of Booleans
;;Returns: A List with all the boolean values reversed

(define lst_of_booleans (list true false true false true))

(define (neg-list lst)
  (cond 
    [(empty? lst) empty]
    [(cons? lst) (cons (false? (first lst)) (neg-list (rest lst)))]))

(neg-list lst_of_booleans)
(neg-list (list false))
(neg-list empty)
